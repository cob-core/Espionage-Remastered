sabotage_recruitment = {
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		nationalism
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}	
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
	}
	
	pact = {	
		cost = 100
	
		is_two_sided_pact = no
		show_in_outliner = yes
		
		
		monthly_effect = {
			custom_tooltip = {
				text = sabotage_recruitment_monthly_effect
				random_list = { 
					35 = {}
					1 = { 
						root = {
							trigger_event = {
					            id = adv_dip_event.41
					        }
					    }
					}
				}								
			}
		} 	
	
		should_invalidate = {
			OR = {		
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				in_default = yes
			}
		}
	}
	
	ai = {

		propose_score = {
			value = 10
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = rivalry
				}
				relations:root <= relations_threshold:cold									
			}
			country_rank >= rank_value:great_power
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}										
					}
				}						
			}					
		}			
	}
}
