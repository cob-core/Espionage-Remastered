election_interference = {
	groups = {
		general
	}
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		mass_communication
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			OR = {
				scope:target_country = {
					any_interest_group = {
						has_party = yes
					}
				}
			}
		}
	institution_investment_level = {
		institution = institution_espionage
		value >= 4
	}
}
	
	possible = {
		is_revolutionary = no
		in_default = no
		scope:target_country = {
			any_interest_group = {
				has_party = yes
			}
		}
	}
	
	pact = {
		cost = 250
	
		income_transfer = 0.05
		max_paying_country_income_to_transfer = 1
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		income_transfer_to_pops = {
			allow_non_fully_accepted = no
			upper_strata_pops = {
				value = 0.8
				max = 1
				min = 0
			}
			middle_strata_pops = {
				value = 0.2
				max = 1
				min = 0				
			}
			lower_strata_pops = {
				value = 0
				max = 1
				min = 0				
			}
		}
		
		monthly_effect = {
			custom_tooltip = {
				text = election_interference_monthly_effect
				random_list = { 
					2 = {}
					1 = {
						if = {
							limit = {
								scope:target_country = {
									in_election_campaign = yes
								}
							}
							root = {
								trigger_event = {
									id = adv_dip_event.81
								}
							}
						}
					}
				}
			}
		}
	
		should_invalidate = {
			OR = {		
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				NOT = {
					scope:target_country = {
						any_interest_group = {
							has_party = yes
						}
					}
				}
				in_default = yes
				NOT {
					has_diplomatic_relevance = scope:target_country
				}
			}
		}
	}
	
	ai = {

		propose_score = {
			value = 5
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = rivalry
				}
				relations:root <= relations_threshold:cold
			}
			country_rank >= rank_value:major_power
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}										
					}
				}						
			}					
		}			
	}
}
