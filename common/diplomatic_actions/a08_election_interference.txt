		}
	}
}
	
	possible = {
		is_revolutionary = no
		in_default = no
	}
	
	pact = {	
		cost = 250

		max_paying_country_income_to_transfer = 1

			allow_non_fully_accepted = no

				value = 1

				min = 0
			}
			middle_strata_pops = {
				value = 0
				max = 1
				min = 0				
			}
			lower_strata_pops = {
				value = 0
				max = 1
				min = 0				
			}
		}
		
		
		monthly_effect = {
			custom_tooltip = {

					            id = adv_dip_event.81

			}

			}
		}

				AND = {

				}
				NOT = {
					scope:target_country = {

						}
