steal_technology = {
    groups = {
        general
    }
    requires_approval = no
    should_notify_third_parties = yes

    unlocking_technologies = {
        rationalism
    }

    selectable = {
        NOT = { is_country_type = decentralized }
        is_revolutionary = no
    }
    
    potential = {
        NOT = { is_country_type = decentralized }
        scope:target_country = {
            NOT = { is_country_type = decentralized }
        }
        institution_investment_level = {
            institution = institution_espionage
            value >= 1
        }
        scope:target_country = {
			country_rank > rank_value:minor_power
		}
        has_game_rule = enable_tech_stealing
    }
    
    possible = {
        is_revolutionary = no
        in_default = no
        NOT = { has_war_with = scope:target_country }
		scope:target_country.techs_researched > root.techs_researched
        has_diplomatic_relevance = scope:target_country
    }
    
    pact = {
        cost = 150
    
        is_two_sided_pact = no
        show_in_outliner = yes
        can_be_used_in_sway_offers = yes
        sway_maneuvers_cost = 15
        
        first_modifier = {
        }

        second_modifier = {
        }

        monthly_effect = {
            custom_tooltip = {
                text = steal_technology_monthly_effect
                random_list = {
                    10 = {}
                    1 = { 
                        root = {
                            trigger_event = {
                                id = adv_dip_event.91
                            }
                        }
                    }
                }
            }
        }
    
        should_invalidate = {
            OR = {
                scope:target_country.techs_researched <= root.techs_researched
                has_war_with = scope:target_country
                is_country_type = decentralized
                scope:target_country = { is_country_type = decentralized }
                in_default = yes
                NOT = {
                    scope:target_country = {
                        any_technology = {
                            is_researched = yes
                            NOT = { root.has_technology_researched = this }
                        }
                    }
                }
            }
        }
    }
    
    ai = {
        max_influence_spending_fraction = 0.15
    
        propose_score = {
            value = 100
        }
    
        will_propose = {
            scaled_debt <= 0
            is_at_war = no
            scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = rivalry
				}
				relations:root <= relations_threshold:cold
			}
        }
        
        will_break = { 
            OR = {        
                scaled_debt > 0.25
                is_at_war = yes
            }
        }            
    }
}
