cultural_exchange = {
	groups = {
		general
	}
	requires_approval = yes
	should_notify_third_parties = no

	unlocking_technologies = {
		empiricism
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		country_rank >= rank_value:great_power
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		institution_investment_level = {
			institution = institution_espionage
			value >= 1
		}
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = { has_war_with = scope:target_country }
		custom_tooltip = {
			text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS_NOT
			scope:target_country.relations:root > relations_threshold:poor
		}
		scope:target_country = {
			country_rank < rank_value:great_power
		}
		has_diplomatic_relevance = scope:target_country
	}
	
	pact = {	
		cost = 150
	
		is_two_sided_pact = yes
		show_in_outliner = yes
		can_be_used_in_sway_offers = yes
		sway_maneuvers_cost = 10

		relations_progress_per_day = 13

		income_transfer = 0.05
		max_paying_country_income_to_transfer = 1
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		income_transfer_to_pops = {
			allow_non_fully_accepted = no
			upper_strata_pops = {
				value = 1
				max = 1
				min = 0
			}
			middle_strata_pops = {
				value = 0
				max = 1
				min = 0				
			}
			lower_strata_pops = {
				value = 0
				max = 1
				min = 0				
			}
		}

		second_modifier = {
			country_pact_leverage_generation_add = 200
		}

		monthly_effect = {
			custom_tooltip = {
				text = send_advisors_monthly_effect
				random_list = { 
					16 = {}
					1 = { 
						root = {
							trigger_event = {
					            id = adv_dip_event.63
					        }
					    }
					}
				}								
			}
		} 	
	
		is_about_to_auto_break = {
            scope:target_country.relations:root <= -17 # this: poor + 3
        }

		should_auto_break = {
			custom_tooltip = {
				text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS

                # make sure to update is_about_to_auto_break when changing this:
				scope:target_country.relations:root <= relations_threshold:poor
			}
		}		
	
		should_invalidate = {
			OR = {
				has_war_with = scope:target_country				
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				in_default = yes
			}
		}
	}
	
	ai = {
		max_influence_spending_fraction = 0.10
	
		propose_score = {
			value = 10
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			scope:target_country = { is_subject_of = root }
			scope:target_country = {
				country_rank < rank_value:major_power
			}
			OR = {
				scope:target_country = { is_subject_of = root }
				has_attitude = {
					who = scope:target_country
					attitude = domineering
				}
				has_attitude = {
					who = scope:target_country
					attitude = cooperative
				}
				has_attitude = {
					who = scope:target_country
					attitude = conciliatory
				}
			}	
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
			}
		}

		accept_score = {
			value = 0
			add = {
				desc = "LOW_RATING_OPINION"
				if = {
					limit = {
						country_rank < rank_value:major_power
					}
					value = 50
				}
			}
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
				if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = genial
							}
							has_attitude = {
								who = scope:actor
								attitude = protective
							}
						}
					}
					value = 100
				}
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = cooperative
							}
							has_attitude = {
								who = scope:actor
								attitude = conciliatory
							}							
						}
					}
					value = 50
				}
				else_if = {
					limit = {
						has_attitude = {
							who = scope:actor
							attitude = domineering
						}
					}
					value = 25
				}
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:actor
								attitude = antagonistic
							}
							has_attitude = {
								who = scope:actor
								attitude = belligerent
							}
						}
					}
					value = -100
				}
				else = {
					value = -25
				}
			}
		}

		accept_break_score = {
			value = 10
		}
	}
}
