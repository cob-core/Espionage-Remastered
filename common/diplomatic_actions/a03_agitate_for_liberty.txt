agitate_for_liberty = {
	groups = {
		general
	}
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		nationalism
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			is_subject = yes
			NOT = { is_subject_of = root}
		}	
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = {
			scope:target_country.overlord = {
				has_diplomatic_pact = {
					who = root
					type = humiliation
				}								
			}
		}
		NOT = {
			scope:target_country.overlord = {
				has_truce_with = scope:target_country							
			}
		}
		institution_investment_level = {
			institution = institution_espionage
			value >= 3
		}
	}
	
	pact = {	
		cost = 200
	
		is_two_sided_pact = no
		show_in_outliner = yes
		income_transfer = 0.75
		max_paying_country_income_to_transfer = 1
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		relations_progress_per_day = 13
		
		monthly_effect = {
			custom_tooltip = {
				text = agitate_for_liberty_monthly_effect
				random_list = { 
					35 = {}
					1 = { 
						root = {
							trigger_event = {
					            id = adv_dip_event.31
					        }
					    }
					}
				}								
			}
		} 	
	
		should_invalidate = {
			OR = {		
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				scope:target_country = { is_subject = no }
				in_default = yes
				scope:target_country.overlord = {
					has_diplomatic_pact = {
						who = root
						type = humiliation
					}								
				}
				scope:target_country.overlord = {
					has_truce_with = scope:target_country							
				}
			}
		}
	}
	
	ai = {
		propose_score = {
			value = 10
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			scope:target_country.overlord = {
				has_diplomatic_pact = {
					who = root
					type = rivalry
				}
				relations:root <= relations_threshold:cold									
			}
			scope:target_country = {
				total_population > 1000000
			}
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					scope:target_country.overlord = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}
					}
				}
				NOT = {
					scope:target_country.overlord = {
						relations:root <= relations_threshold:cold											
					}
				}
				AND = {
					scope:target_country = {
						relations:overlord <= relations_threshold:cold
					}
					has_truce_with = scope:target_country.overlord
				}
			}					
		}			
	}
}
