support_nationalists = {
	groups = {
		general
	}
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		nationalism
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		country_rank >= rank_value:great_power
		institution_investment_level = {
			institution = institution_espionage
			value >= 4
		}
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = {
			scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = humiliation
				}								
			}
		}
		has_diplomatic_relevance = scope:target_country
	}
	
	pact = {	
		cost = 200
	
		is_two_sided_pact = no
		show_in_outliner = yes
		
		income_transfer = 0.1
		max_paying_country_income_to_transfer = 1
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		monthly_effect = {
			custom_tooltip = {
				text = support_nationalists_monthly_effect
				random_list = { 
					2 = {} #35
					1 = { 
						root = {
							trigger_event = {
					            id = adv_dip_event.21
					        }
					    }
					}
				}								
			}
		} 	
	
		should_invalidate = {
			OR = {		
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				in_default = yes
				scope:target_country = {
					has_diplomatic_pact = {
						who = root
						type = humiliation
					}								
				}
			}
		}
	}
	
	ai = {

		propose_score = {
			value = 10
		}
	
		will_propose = { 
			scaled_debt <= 0
			is_at_war = no
			scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = rivalry
				}
				relations:root <= relations_threshold:cold									
			}
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}										
					}
				}
				NOT = {
					scope:target_country = {
						relations:root <= relations_threshold:cold											
					}
				}						
			}					
		}			
	}
}
