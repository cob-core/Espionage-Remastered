fund_piracy = {
	groups = {
		general
	}
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		admiralty
	}
	
	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		country_rank >= rank_value:great_power
		institution_investment_level = {
			institution = institution_espionage
			value >= 4
		}
	}
	
	possible = {
		is_revolutionary = no
		in_default = no
		NOT = {
			scope:target_country = {
				has_diplomatic_pact = {
					who = root
					type = humiliation
				}
			}
		}
		has_diplomatic_relevance = scope:target_country
		
	}
	
	pact = {	
		cost = 200
	
		is_two_sided_pact = no
		show_in_outliner = yes
		
		income_transfer = 0.025
		max_paying_country_income_to_transfer = 1
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		income_transfer_to_pops = {
			allow_non_fully_accepted = yes
			upper_strata_pops = {
				value = 0
				max = 1
				min = 0
			}
			middle_strata_pops = {
				value = 0.2
				max = 1
				min = 0				
			}
			lower_strata_pops = {
				value = 0.8
				max = 1
				min = 0				
			}
		}

		monthly_effect = {
			custom_tooltip = {
				text = fund_piracy_monthly_effect
				random_list = { 
					16 = {}
					1 = { 
						root = {
							trigger_event = {
					            id = adv_dip_event.101
					        }
					    }
					}
				}
			}
		} 	
	
		should_invalidate = {
			OR = {		
				is_country_type = decentralized
				scope:target_country = { is_country_type = decentralized }
				in_default = yes
				scope:target_country = {
					has_diplomatic_pact = {
						who = root
						type = humiliation
					}								
				}
			}
		}
	}
	
	ai = {

		propose_score = {
			value = 10
			if = {
				limit = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}
					}
				}
			}
		}
	
		will_propose = { 
			scaled_debt <= 0
				relations:root <= relations_threshold:cold
			}
		}
		
		will_break = { 
			OR = {		
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					is_at_war = yes
				}
				NOT = {
					scope:target_country = {
						relations:root <= relations_threshold:cold											
					}
				}						
			}					
		}			
	}
}
